 static struct xt_match xt_hashlimit[] __read_mostly = {
    {
        .name       = "hashlimit",
        .family     = AF_INET,
        .match      = hashlimit_match,
        .matchsize  = sizeof(struct xt_hashlimit_info),
#ifdef CONFIG_COMPAT
        .compatsize = sizeof(struct compat_xt_hashlimit_info),
        .compat_from_user = compat_from_user,
        .compat_to_user = compat_to_user,
#endif
        .checkentry = hashlimit_checkentry,
        .destroy    = hashlimit_destroy,
        .me     = THIS_MODULE
    },

  1. hashlimit_match
    • 思路：从sk_buff结构的skb中获取IP和端口信息，存放到struct dsthash_dst中，然后在结构为struct xt_hashlimit_htable的hinfo中调用dsthash_find函数。将skb中获得的最新的IP进行hash，然后找hinfo中对应的struct dsthash_dst 整个结构进行cmp，相同则找到，不同则没有找到。
  2. hashlimit_checkentry
  3. hashlimit_destry
    • 思路：删除一个xt_hashlimit_info节点。
