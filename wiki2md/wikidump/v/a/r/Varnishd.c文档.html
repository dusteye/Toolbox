NAME 

    varnishd — HTTP accelerator daemon

SYNOPSIS 

    varnishd [-a address[:port]] [-b host[:port]] [-d] [-F] [-f config]
             [-g group] [-h type[,options]] [-l shmlogsize] [-n name]
             [-P file] [-p param=value] [-s type[,options]]
             [-T address[:port]] [-t ttl] [-u user] [-V]
             [-w min[,max[,timeout]]]
